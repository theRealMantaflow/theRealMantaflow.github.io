    const dirLightTopF = new THREE.DirectionalLight(0xffffff, 2);

    // Crank up the resolution for crisp PCB component shadows
    dirLightTopF.shadow.mapSize.width = 2048;
    dirLightTopF.shadow.mapSize.height = 2048;

    // Define the box area where shadows are calculated
    dirLightTopF.shadow.camera.left = -15;
    dirLightTopF.shadow.camera.right = 15;
    dirLightTopF.shadow.camera.top = 15;
    dirLightTopF.shadow.camera.bottom = -15;

    // Crucial for flat PCBs: prevents shadow glitching on the board surface
    dirLightTopF.shadow.bias = -0.0005;

    camera.add(dirLightTopF);
